  export SERVICE_IP=$(kubectl get svc {{ include "ai-knowledge-agent.fullname" . }} \

    -o jsonpath='{.status.loadBalancer.ingress[0].hostname}')
  echo http://$SERVICE_IP:{{ .Values.service.port }}

Access the application via port-forward:
  kubectl port-forward svc/{{ include "ai-knowledge-agent.fullname" . }} \
    -n {{ .Release.Namespace }} 8501:{{ .Values.service.port }}

  Then open: http://localhost:8501

  kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=agent -f
